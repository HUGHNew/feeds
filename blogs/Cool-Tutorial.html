# HelloWorld

class Main inherits IO{
    hw:String <- "HelloWorld\n"; -- initialization or assignment
    hello():String{"Hello"}; -- define a method
    main():SELF_TYPE{ -- return type:Main
        out_string(hw)
    };
};
