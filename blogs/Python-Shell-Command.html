# subprocess.Popenopen in new window

此类可以创造和管理执行的进程

import subprocess
process = subprocess.Popen(['echo', 'More output'],
                     stdout=subprocess.PIPE, 
                     stderr=subprocess.PIPE) # 指定为 pipe 通过 communicate 读取
stdout, stderr = process.communicate() # 类型都是 bytes
